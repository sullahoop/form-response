function onFormSubmit(e) {
  const values = e.namedValues;

  const payload = {
    timestamp: new Date().toISOString(),
    answers: values
  };

  UrlFetchApp.fetch("https://your-server.com/api/form-response", {
    method: "post",
    contentType: "application/json",
    payload: JSON.stringify(payload)
  });
}
